--- CARA MENGGUNAKAN UTAMA (API) ---

1. Inisialisasi Neural Network
   from mlpnn import MultiLayerPERCEPTRONNeuralNetwork
   
   # Membuat MLP dengan 4 input, 5 hidden, 1 output
   mlp = MultiLayerPERCEPTRONNeuralNetwork(
       layers=[4, 5, 1],
       type_of_activation_function="sigmoid", # atau "relu"
       seed=42,
       learning_rate=0.1
   )

2. Melatih Neural Network (Training)
   # X_train: array numpy (n_samples, n_features)
   # y_train: array numpy (n_samples, n_outputs)
   
   mlp.train(X_train, y_train, epochs=1000)

3. Melakukan Prediksi
   outputs = mlp.predict(X_new_data)
